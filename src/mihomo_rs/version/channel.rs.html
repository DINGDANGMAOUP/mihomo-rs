5#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
6pub enum Channel {
7    Stable,
8    Beta,
9    Nightly,
10}
11
12impl Channel {
13    pub fn as_str(&self) -> &str {
14        match self {
15            Channel::Stable => "stable",
16            Channel::Beta => "beta",
17            Channel::Nightly => "nightly",
18        }
19    }
20}
21
22impl FromStr for Channel {
23    type Err = String;
24
25    fn from_str(s: &str) -> std::result::Result<Self, Self::Err> {
26        match s.to_lowercase().as_str() {
27            "stable" => Ok(Channel::Stable),
28            "beta" => Ok(Channel::Beta),
29            "nightly" | "alpha" => Ok(Channel::Nightly),
30            _ => Err(format!("Invalid channel: {}", s)),
31        }
32    }
33}

35#[derive(Debug, Clone, Serialize, Deserialize)]
36pub struct ChannelInfo {
37    pub channel: Channel,
38    pub version: String,
39    pub release_date: String,
40}

42pub async fn fetch_latest(channel: Channel) -> Result<ChannelInfo> {
43    let url = match channel {
44        Channel::Stable => "https://api.github.com/repos/MetaCubeX/mihomo/releases/latest",
45        Channel::Beta => {
46            "https://api.github.com/repos/MetaCubeX/mihomo/releases?per_page=1&prerelease=true"
47        }
48        Channel::Nightly => "https://api.github.com/repos/MetaCubeX/mihomo/releases?per_page=1",
49    };
50
51    let client = reqwest::Client::new();
52    let resp = client
53        .get(url)
54        .header("User-Agent", "mihomo-rs")
55        .send()
56        .await?;
57
58    let data: serde_json::Value = resp.json().await?;
59
60    let (version, date) = if channel == Channel::Stable {
61        let tag = data["tag_name"].as_str().unwrap_or("").to_string();
62        let date = data["published_at"].as_str().unwrap_or("").to_string();
63        (tag, date)
64    } else {
65        let empty_vec = vec![];
66        let releases = data.as_array().unwrap_or(&empty_vec);
67        if let Some(release) = releases.first() {
68            let tag = release["tag_name"].as_str().unwrap_or("").to_string();
69            let date = release["published_at"].as_str().unwrap_or("").to_string();
70            (tag, date)
71        } else {
72            return Err(crate::core::MihomoError::Version(
73                "No releases found".to_string(),
74            ));
75        }
76    };
77
78    Ok(ChannelInfo {
79        channel,
80        version,
81        release_date: date,
82    })
83}

94pub async fn fetch_releases(limit: usize) -> Result<Vec<ReleaseInfo>> {
95    let client = reqwest::Client::new();
96    let resp = client
97        .get(format!(
98            "https://api.github.com/repos/MetaCubeX/mihomo/releases?per_page={}",
99            limit
100        ))
101        .header("User-Agent", "mihomo-rs")
102        .send()
103        .await?;
104
105    if !resp.status().is_success() {
106        return Err(crate::core::MihomoError::Version(format!(
107            "GitHub API error: {}",
108            resp.status()
109        )));
110    }
111
112    let releases: Vec<ReleaseInfo> = resp.json().await?;
113    Ok(releases)
