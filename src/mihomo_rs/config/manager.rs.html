29    pub async fn load(&self, profile: &str) -> Result<String> {
30        let path = self.config_dir.join(format!("{}.yaml", profile));
31        if !path.exists() {
32            return Err(MihomoError::NotFound(format!(
33                "Profile '{}' not found",
34                profile
35            )));
36        }
37
38        let content = fs::read_to_string(&path).await?;
39        Ok(content)
40    }

42    pub async fn save(&self, profile: &str, content: &str) -> Result<()> {
43        fs::create_dir_all(&self.config_dir).await?;
44
45        serde_yaml::from_str::<serde_yaml::Value>(content)?;
46
47        let path = self.config_dir.join(format!("{}.yaml", profile));
48        fs::write(&path, content).await?;
49
50        Ok(())
51    }

53    pub async fn list_profiles(&self) -> Result<Vec<Profile>> {
54        if !self.config_dir.exists() {
55            return Ok(vec![]);
56        }
57
58        let current = self.get_current().await.ok();
59        let mut profiles = vec![];
60
61        let mut entries = fs::read_dir(&self.config_dir).await?;
62        while let Some(entry) = entries.next_entry().await? {
63            let path = entry.path();
64            if path.extension().and_then(|s| s.to_str()) == Some("yaml") {
65                let name = path
66                    .file_stem()
67                    .and_then(|s| s.to_str())
68                    .unwrap_or("")
69                    .to_string();
70                let active = current.as_ref() == Some(&name);
71                profiles.push(Profile::new(name, path, active));
72            }
73        }
74
75        profiles.sort_by(|a, b| a.name.cmp(&b.name));
76        Ok(profiles)
77    }

99    pub async fn set_current(&self, profile: &str) -> Result<()> {
100        let path = self.config_dir.join(format!("{}.yaml", profile));
101        if !path.exists() {
102            return Err(MihomoError::NotFound(format!(
103                "Profile '{}' not found",
104                profile
105            )));
106        }
107
108        fs::create_dir_all(self.settings_file.parent().unwrap()).await?;
109
110        let mut config = if self.settings_file.exists() {
111            let content = fs::read_to_string(&self.settings_file).await?;
112            toml::from_str(&content).unwrap_or_else(|_| toml::Value::Table(toml::map::Map::new()))
113        } else {
114            toml::Value::Table(toml::map::Map::new())
115        };
116
117        if let toml::Value::Table(ref mut table) = config {
118            let default_table = table
119                .entry("default".to_string())
120                .or_insert_with(|| toml::Value::Table(toml::map::Map::new()));
121
122            if let toml::Value::Table(ref mut default) = default_table {
123                default.insert(
124                    "profile".to_string(),
125                    toml::Value::String(profile.to_string()),
126                );
127            }
128        }
129
130        let content = toml::to_string(&config)
131            .map_err(|e| MihomoError::Config(format!("Failed to serialize config: {}", e)))?;
132        fs::write(&self.settings_file, content).await?;
133
134        Ok(())
135    }

137    pub async fn get_current(&self) -> Result<String> {
138        if !self.settings_file.exists() {
139            return Ok("default".to_string());
140        }
141
142        let content = fs::read_to_string(&self.settings_file).await?;
143        let config: toml::Value = toml::from_str(&content)
144            .map_err(|e| MihomoError::Config(format!("Invalid config: {}", e)))?;
145
146        Ok(config
147            .get("default")
148            .and_then(|d| d.get("profile"))
149            .and_then(|p| p.as_str())
150            .unwrap_or("default")
151            .to_string())
152    }

159    /// Ensure a default config file exists, create one if it doesn't
160    pub async fn ensure_default_config(&self) -> Result<()> {
161        let profile = self.get_current().await?;
162        let path = self.config_dir.join(format!("{}.yaml", profile));
163
164        if !path.exists() {
165            log::info!("Default config '{}' not found, creating...", profile);
166
167            let port = find_available_port(9090).ok_or_else(|| {
168                MihomoError::Config("No available ports found in range 9090-9190".to_string())
169            })?;
170
171            let default_config = format!(
172                r#"# mihomo configuration
173port: 7890
174socks-port: 7891
175allow-lan: false
176mode: rule
177log-level: info
178external-controller: 127.0.0.1:{}
179"#,
180                port
181            );
182
183            self.save(&profile, &default_config).await?;
184            log::info!("Created default config at: {}", path.display());
185        }
186
187        Ok(())
188    }

190    pub async fn get_external_controller(&self) -> Result<String> {
191        let profile = self.get_current().await?;
192        log::debug!("Reading external-controller from profile: {}", profile);
193
194        let content = self.load(&profile).await?;
195        let config: serde_yaml::Value = serde_yaml::from_str(&content)?;
196
197        let controller = config
198            .get("external-controller")
199            .and_then(|v| v.as_str())
200            .unwrap_or("127.0.0.1:9090");
201
202        let url = if controller.starts_with(':') {
203            format!("http://127.0.0.1{}", controller)
204        } else if controller.starts_with("http://") || controller.starts_with("https://") {
205            controller.to_string()
206        } else {
207            format!("http://{}", controller)
208        };
209
210        log::debug!("External controller URL: {}", url);
211        Ok(url)
212    }

214    /// Ensure external-controller is configured in the current profile
215    /// If not present or port is occupied, add/update it with an available port
216    pub async fn ensure_external_controller(&self) -> Result<String> {
217        let profile = self.get_current().await?;
218        let content = self.load(&profile).await?;
219        let mut config: serde_yaml::Value = serde_yaml::from_str(&content)?;
220
221        let needs_update = match config.get("external-controller").and_then(|v| v.as_str()) {
222            Some(controller) => {
223                // Parse the port from the controller address
224                let addr = if controller.starts_with(':') {
225                    format!("127.0.0.1{}", controller)
226                } else {
227                    controller.to_string()
228                };
229
230                match parse_port_from_addr(&addr) {
231                    Some(port) => {
232                        if !is_port_available(port) {
233                            log::warn!("Port {} is occupied, finding alternative", port);
234                            true
235                        } else {
236                            false
237                        }
238                    }
239                    None => {
240                        log::warn!("Invalid external-controller format: {}", controller);
241                        true
242                    }
243                }
244            }
245            None => {
246                log::info!("external-controller not found in config, adding default");
247                true
248            }
249        };
250
251        if needs_update {
252            let port = find_available_port(9090).ok_or_else(|| {
253                MihomoError::Config("No available ports found in range 9090-9190".to_string())
254            })?;
255
256            let controller_addr = format!("127.0.0.1:{}", port);
257            log::info!("Setting external-controller to {}", controller_addr);
258
259            if let serde_yaml::Value::Mapping(ref mut map) = config {
260                map.insert(
261                    serde_yaml::Value::String("external-controller".to_string()),
262                    serde_yaml::Value::String(controller_addr.clone()),
263                );
264            }
265
266            let updated_content = serde_yaml::to_string(&config)?;
267            self.save(&profile, &updated_content).await?;
268
269            Ok(format!("http://{}", controller_addr))
270        } else {
271            self.get_external_controller().await
272        }
273    }
